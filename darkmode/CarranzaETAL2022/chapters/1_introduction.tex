\begin{frame}{Literature}
    \begin{itemize}

        {\footnotesize 
        \begin{itemize}
            \item<3->[-] \texthlit{firms}: skill revelation and learning {\scriptsize \citep{kahn2014employer,arcidiacono2010beyond,heath2018firms,macleod2017big}}
            \item<4->[-] \texthlit{workseekers}: inaccurate beliefs and self-correction {\scriptsize \citep{spinnewijn2015unemployed,ahn2019improving,banerjee2020spatial,altmann2018learning,belot2019providing}}

            \uncover<7->{but could \textbf{\color{fuzzywuzzy!65!white}\underline{backfire}} without bilateral information sharing {\scriptsize \citep{alonso2018recruiting}}} 
        \end{itemize}
        }
        \vspace*{5pt}
