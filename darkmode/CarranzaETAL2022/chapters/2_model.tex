                \small 
                \item under-informed \textcolor{fuzzywuzzy!65!white}{\textbf{firms}}
                
                \uncover<2->{output:
                $$ q\underbrace{\cdot V_{j,j}}_{\text{optimal}} + (1-q)\cdot \underbrace{V_{i,j}}_{\text{suboptimal}} $$}
                \uncover<3->{wage under \textcolor{fuzzywuzzy!65!white}{\textbf{concavity}}:
                $$ W_j \leq qW_{j,j} + (1-q)\cdot W_{i,j} $$}
            \end{itemize}
            
        \end{column}
    \end{columns}
    \uncover<4>{
        \texthlit{prediction}
        \begin{itemize}
        \small
        \item[-] lower wages conditional on employment, lower employment
        \item[-] costly screening technology cannot fully mitigate this inefficiency
    \end{itemize}}
\end{frame}

                \begin{center}
                    \begin{tabular}{lccc}
                    
                    & &\multicolumn{2}{c}{workseekers} \\
                   & & $W_1$ & $W_2$ \\
                    \hline
                    \multirow{2}{*}{jobs} & $J_1$ & ${\boxed{P_{1,1}\cdot U(\underbrace{W_{1,1}}_{\textcolor<2->{fuzzywuzzy!65!white}{\leq V_{1,1}}})-C}}$ & $P_{2,1}\cdot U(\underbrace{W_{2,1}}_{\leq V_{2,1}})-C$ \\
                    & $J_2$ & $P_{1,2}\cdot U(\underbrace{W_{1,2}}_{\textcolor<3>{fuzzywuzzy!65!white}{\leq V_{1,2}}})-C$ & ${\boxed{P_{2,2}\cdot U(\underbrace{W_{2,2}}_{\textcolor<2>{fuzzywuzzy!65!white}{\leq V_{2,2}}})-C}}$
                    \end{tabular}
                \end{center}
            \end{table}

        \end{column}
    
        \begin{column}{0.35\textwidth}
            \begin{itemize}
                \small 
                \item<2-> \textcolor{fuzzywuzzy!65!white}{\textbf{horizontal}} differentiation: $$V_{i,i}>V_{i,j},V_{j,j}>V_{j,i}$$
                \item<3-> \textcolor{fuzzywuzzy!65!white}{\textbf{vertical}} differentiation: $$V_{i,\cdot}>V_{j,\cdot}$$
            \end{itemize}
            
        \end{column}
    \end{columns}
    \uncover<4>{
        \texthlit{prediction}
        \begin{itemize}
        \small
        \item[-] \textcolor{fuzzywuzzy!65!white}{\textbf{horizontal}} differentiation: lower wages and employment for all types
        \item[-] \textcolor{fuzzywuzzy!65!white}{\textbf{vertical}} differentiation: \textit{\underline{underprice}} efficient types, \textit{\underline{overprice}} inefficient types
    \end{itemize}}
\end{frame}
